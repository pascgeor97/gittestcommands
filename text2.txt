git init // adds git in a folder
git add "name" or git add . //add everything in stage
git rm --cached -r . //remove everything from stage
git status // check the status 
git commit -m "text" // commit the files
git diff "name" //checks the difference in a file
git checkout "name" // roll back all the changes that happen to the specific file
touch .gitignore // a file that ignires git
